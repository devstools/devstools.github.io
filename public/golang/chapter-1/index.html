Function In Struct

package main

import "fmt"

type Student struct {
	id int
	name string
	grade int
}

func (s Student) Display(){
	fmt.Println(s)
}

func main() {
    var s2 = Student{2, "Steffi", 10} 
    s2.Display()

    var s3 = Student{id:3, name:"Steve" } 
    s3.Display() //output: {3 Steve 0}
}
